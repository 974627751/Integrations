# @datafire/slideroom
SlideRoom provides a RESTful API for exporting data out of your organization's SlideRoom account.

## Operation: Applicant_DeleteAttributesV2
Deletes a custom attribute for an applicant.

### Input Schema
```json
{
  "type": "object",
  "properties": {
    "email": {
      "type": "string",
      "description": "The email address of the applicant."
    },
    "name": {
      "type": "string",
      "description": "The name of the attribute to be deleted."
    },
    "pool": {
      "type": "string",
      "enum": [
        "Standard",
        "CommonAppSDS"
      ]
    },
    "commonAppYear": {
      "type": "integer",
      "format": "int32"
    }
  },
  "additionalProperties": false,
  "required": [
    "email",
    "name"
  ]
}
```
### Output Schema
```json
{
  "type": "string"
}
```

## Operation: Application_RequestExportV2

            PDF and ZIP exports are available in the Advanced Plan.

### Input Schema
```json
{
  "type": [
    "object",
    "null"
  ],
  "properties": {
    "format": {
      "type": "string",
      "enum": [
        "csv",
        "tsv",
        "txt",
        "tab",
        "xlsx",
        "pdf",
        "zip",
        "json"
      ]
    },
    "roundType": {
      "type": "string",
      "enum": [
        "Assigned",
        "Current",
        "Named",
        "All"
      ]
    },
    "roundName": {
      "type": "string"
    },
    "tab.export": {
      "type": "string"
    },
    "pdf.includeForms": {
      "type": "boolean"
    },
    "pdf.includeReferences": {
      "type": "boolean"
    },
    "pdf.includeMedia": {
      "type": "boolean"
    },
    "pdf.includeApplicantAttachments": {
      "type": "boolean"
    },
    "pdf.includeOrganizationAttachments": {
      "type": "boolean"
    },
    "pdf.includeRatings": {
      "type": "boolean"
    },
    "pdf.includeFullPageMedia": {
      "type": "boolean"
    },
    "pdf.includeHighlights": {
      "type": "boolean"
    },
    "pdf.includeComments": {
      "type": "boolean"
    },
    "pdf.includeCommonApp": {
      "type": "boolean"
    },
    "zip.originalMedia": {
      "type": "boolean"
    },
    "zip.includeForms": {
      "type": "boolean"
    },
    "zip.includeReferences": {
      "type": "boolean"
    },
    "zip.includeMedia": {
      "type": "boolean"
    },
    "zip.includeApplicantAttachments": {
      "type": "boolean"
    },
    "zip.includeOrganizationAttachments": {
      "type": "boolean"
    },
    "zip.includeRatings": {
      "type": "boolean"
    },
    "zip.includeComments": {
      "type": "boolean"
    },
    "zip.includeCommonApp": {
      "type": "boolean"
    },
    "delivery.account": {
      "type": "string"
    },
    "delivery.folder": {
      "type": "string"
    },
    "since": {
      "type": "integer",
      "format": "int32"
    },
    "pool": {
      "type": "string",
      "enum": [
        "All",
        "Current",
        "Archived",
        "CommonAppSDS"
      ]
    },
    "status": {
      "type": "string",
      "enum": [
        "All",
        "InProgress",
        "Submitted"
      ]
    },
    "searchName": {
      "type": "string"
    },
    "email": {
      "type": "string"
    }
  },
  "additionalProperties": false
}
```

```json
}
```
